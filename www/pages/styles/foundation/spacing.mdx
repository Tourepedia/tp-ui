import Layout from "../../../components/StylesLayout"
import { theme } from "@tourepedia/ui"

export const meta = {
    title: "Spacing",
    description: "Spacing"
};

export default Layout

The spacing scale is a `base-.25rem` scale. So the scale of 1 means a spacing of `0.25em`. These spacing units are
shared between `padding`, `margin`, `height` and `width` utils.

export function Spacing () {
  const { spacing } = theme
  return <table className="table max-w-sm table-bordered table-striped"><thead><tr><th style={{ textAlign: "right" }}>Scale</th><th>Value</th></tr></thead><tbody>{Object.keys(spacing).map((scale) => {
    const value = spacing[scale]
    if (typeof value === "string") {
    return <tr key={scale}><td style={{ textAlign: "right" }}>{scale}</td><td>{value}</td></tr>
    }
    return null
    })}</tbody>
  </table>
}

<Spacing />
