Dialog

Dialog component for React built on Tourepedia Design System

Installation

npm install --save @tourepedia/dialog

Usage

import Dialog, { useDialog } from "@tourepedia/dialog"
function App () {
  const [isOpen, open, close] = useDialog(false)
  return <>
    <Dialog open={isOpen} onClose={close}>
      Dialog Content Here
    </Dialog>
    <Button onClick={open}>Open</Button>
  </>
}
function () {
  const [isOpen, open, close] = useDialog(false)
  return <div>
    <Dialog open={isOpen} onClose={close}>
      <Dialog.Header>
        <Dialog.Title>Dialog Title</Dialog.Title>
      </Dialog.Header>
      <Dialog.Body>
        <h2>Here is some stuff inside the dialog</h2>
        <p>Some more stuff</p>
      </Dialog.Body>
      <Dialog.Footer>
        <Button primary onClick={close}>OK! Close Now</Button>
      </Dialog.Footer>
    </Dialog>
    <p>Here is a simple Dialog Component</p>
    <Button onClick={open}>Open Dialog</Button>
  </div>
}
